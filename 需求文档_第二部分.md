```javascript
// worker/index.js
export default {
  async fetch(request, env) {
    // CORS预检请求
    if (request.method === 'OPTIONS') {
      return new Response(null, {
        headers: {
          'Access-Control-Allow-Origin': env.ALLOWED_ORIGINS || '*',
          'Access-Control-Allow-Methods': 'POST, OPTIONS',
          'Access-Control-Allow-Headers': 'Content-Type',
          'Access-Control-Max-Age': '86400',
        },
      });
    }

    // 只允许POST请求
    if (request.method !== 'POST') {
      return new Response('Method not allowed', { status: 405 });
    }

    try {
      // 解析请求体
      const { image, format = 'png', size = 'auto' } = await request.json();
      
      if (!image) {
        return new Response(
          JSON.stringify({ error: 'Missing image data' }),
          { status: 400, headers: { 'Content-Type': 'application/json' } }
        );
      }

      // 调用Remove.bg API
      const removeBgResponse = await fetch('https://api.remove.bg/v1.0/removebg', {
        method: 'POST',
        headers: {
          'X-Api-Key': env.REMOVEBG_API_KEY,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          image_file_b64: image,
          size: size,
          format: format,
        }),
      });

      if (!removeBgResponse.ok) {
        const errorText = await removeBgResponse.text();
        throw new Error(`Remove.bg API error: ${removeBgResponse.status} - ${errorText}`);
      }

      // 获取Base64响应
      const result = await removeBgResponse.json();
      
      // 返回处理结果
      return new Response(
        JSON.stringify({
          success: true,
          data: result.data.result_b64,
          credits_charged: result.data.credits_charged,
        }),
        {
          status: 200,
          headers: {
            'Content-Type': 'application/json',
            'Access-Control-Allow-Origin': env.ALLOWED_ORIGINS || '*',
          },
        }
      );

    } catch (error) {
      console.error('Worker error:', error);
      
      return new Response(
        JSON.stringify({
          success: false,
          error: error.message,
          timestamp: new Date().toISOString(),
        }),
        {
          status: 500,
          headers: {
            'Content-Type': 'application/json',
            'Access-Control-Allow-Origin': env.ALLOWED_ORIGINS || '*',
          },
        }
      );
    }
  },
};
```

#### 前端调用实现
```javascript
// public/js/api.js
class QuickBGAPI {
  constructor(workerUrl) {
    this.workerUrl = workerUrl || '/api/remove-bg';
  }

  async removeBackground(imageBase64, options = {}) {
    const payload = {
      image: imageBase64,
      format: options.format || 'png',
      size: options.size || 'auto',
    };

    try {
      const response = await fetch(this.workerUrl, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify(payload),
      });

      const result = await response.json();

      if (!response.ok || !result.success) {
        throw new Error(result.error || 'API request failed');
      }

      return {
        success: true,
        imageData: result.data,
        credits: result.credits_charged,
        timestamp: new Date().toISOString(),
      };

    } catch (error) {
      console.error('API call failed:', error);
      throw error;
    }
  }

  // 图片转Base64（移除前缀）
  imageToBase64(file) {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload = (e) => {
        // 移除 data:image/png;base64, 前缀
        const base64 = e.target.result.split(',')[1];
        resolve(base64);
      };
      reader.onerror = reject;
      reader.readAsDataURL(file);
    });
  }

  // Base64转Blob
  base64ToBlob(base64, mimeType = 'image/png') {
    const byteCharacters = atob(base64);
    const byteArrays = [];

    for (let offset = 0; offset < byteCharacters.length; offset += 512) {
      const slice = byteCharacters.slice(offset, offset + 512);
      const byteNumbers = new Array(slice.length);
      
      for (let i = 0; i < slice.length; i++) {
        byteNumbers[i] = slice.charCodeAt(i);
      }
      
      const byteArray = new Uint8Array(byteNumbers);
      byteArrays.push(byteArray);
    }

    return new Blob(byteArrays, { type: mimeType });
  }
}
```

```javascript
// public/js/app.js
class QuickBGApp {
  constructor() {
    this.api = new QuickBGAPI();
    this.currentFile = null;
    this.processedImage = null;
    this.isProcessing = false;
    
    this.init();
  }

  init() {
    this.bindEvents();
    this.updateUI();
  }

  bindEvents() {
    // 文件选择
    document.getElementById('file-input').addEventListener('change', (e) => {
      this.handleFileSelect(e.target.files[0]);
    });

    // 拖放上传
    const dropZone = document.getElementById('drop-zone');
    dropZone.addEventListener('dragover', (e) => {
      e.preventDefault();
      dropZone.classList.add('dragover');
    });

    dropZone.addEventListener('dragleave', () => {
      dropZone.classList.remove('dragover');
    });

    dropZone.addEventListener('drop', (e) => {
      e.preventDefault();
      dropZone.classList.remove('dragover');
      
      if (e.dataTransfer.files.length > 0) {
        this.handleFileSelect(e.dataTransfer.files[0]);
      }
    });

    // 处理按钮
    document.getElementById('process-btn').addEventListener('click', () => {
      this.processImage();
    });

    // 下载按钮
    document.getElementById('download-btn').addEventListener('click', () => {
      this.downloadImage();
    });

    // 重置按钮
    document.getElementById('reset-btn').addEventListener('click', () => {
      this.reset();
    });
  }

  async handleFileSelect(file) {
    try {
      // 验证文件
      this.validateFile(file);
      
      // 显示加载状态
      this.showLoading('正在加载图片...');
      
      // 保存文件引用
      this.currentFile = file;
      
      // 预览原始图片
      await this.previewOriginalImage(file);
      
      // 更新UI状态
      this.updateUI();
      
      this.hideLoading();
      
    } catch (error) {
      this.showError(error.message);
      console.error('File selection error:', error);
    }
  }

  validateFile(file) {
    const ALLOWED_TYPES = ['image/jpeg', 'image/png', 'image/webp'];
    const MAX_SIZE = 5 * 1024 * 1024; // 5MB

    if (!ALLOWED_TYPES.includes(file.type)) {
      throw new Error('请选择 JPG、PNG 或 WebP 格式的图片');
    }

    if (file.size > MAX_SIZE) {
      throw new Error('图片大小不能超过 5MB');
    }
  }

  async previewOriginalImage(file) {
    return new Promise((resolve) => {
      const reader = new FileReader();
      reader.onload = (e) => {
        const img = new Image();
        img.onload = () => {
          const canvas = document.getElementById('original-canvas');
          const ctx = canvas.getContext('2d');
          
          // 调整画布大小
          const maxWidth = 400;
          const maxHeight = 400;
          let width = img.width;
          let height = img.height;
          
          if (width > maxWidth || height > maxHeight) {
            const ratio = Math.min(maxWidth / width, maxHeight / height);
            width *= ratio;
            height *= ratio;
          }
          
          canvas.width = width;
          canvas.height = height;
          ctx.drawImage(img, 0, 0, width, height);
          
          // 显示画布，隐藏占位符
          document.getElementById('original-placeholder').style.display = 'none';
          canvas.style.display = 'block';
          
          resolve();
        };
        img.src = e.target.result;
      };
      reader.readAsDataURL(file);
    });
  }

  async processImage() {
    if (!this.currentFile || this.isProcessing) {
      return;
    }

    try {
      this.isProcessing = true;
      this.showProgress('正在去除背景...', 0);
      
      // 转换为Base64
      const base64Image = await this.api.imageToBase64(this.currentFile);
      
      // 更新进度
      this.updateProgress(30);
      
      // 调用API
      const result = await this.api.removeBackground(base64Image);
      
      // 更新进度
      this.updateProgress(80);
      
      // 保存处理结果
      this.processedImage = result.imageData;
      
      // 显示处理结果
      await this.previewResultImage(this.processedImage);
      
      // 完成
      this.updateProgress(100);
      setTimeout(() => {
        this.hideProgress();
        this.isProcessing = false;
        this.updateUI();
        this.showSuccess('背景去除成功！');
      }, 500);
      
    } catch (error) {
      this.showError(`处理失败: ${error.message}`);
      console.error('Processing error:', error);
      this.hideProgress();
      this.isProcessing = false;
    }
  }

  async previewResultImage(base64Image) {
    return new Promise((resolve) => {
      const img = new Image();
      img.onload = () => {
        const canvas = document.getElementById('result-canvas');
        const ctx = canvas.getContext('2d');
        
        // 使用与原始图片相同的尺寸
        const originalCanvas = document.getElementById('original-canvas');
        canvas.width = originalCanvas.width;
        canvas.height = originalCanvas.height;
        
        // 绘制透明背景（棋盘格）
        this.drawCheckerboard(ctx, canvas.width, canvas.height);
        
        // 绘制处理后的图片
        ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
        
        // 显示画布，隐藏占位符
        document.getElementById('result-placeholder').style.display = 'none';
        canvas.style.display = 'block';
        
        resolve();
      };
      img.src = `data:image/png;base64,${base64Image}`;
    });
  }

  drawCheckerboard(ctx, width, height) {
    const size = 20;
    const colors = ['#f0f0f0', '#ffffff'];
    
    for (let y = 0; y < height; y += size) {
      for (let x = 0; x < width; x += size) {
        const colorIndex = Math.floor(x / size + y / size) % 2;
        ctx.fillStyle = colors[colorIndex];
        ctx.fillRect(x, y, size, size);
      }
    }
  }

  downloadImage() {
    if (!this.processedImage) {
      this.showError('没有可下载的图片');
      return;
    }

    try {
      const blob = this.api.base64ToBlob(this.processedImage, 'image/png');
      const url = URL.createObjectURL(blob);
      const link = document.createElement('a');
      
      const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
      link.download = `background-removed-${timestamp}.png`;
      link.href = url;
      link.click();
      
      URL.revokeObjectURL(url);
      
    } catch (error) {
      this.showError('下载失败: ' + error.message);
      console.error('Download error:', error);
    }
  }

  reset() {
    this.currentFile = null;
    this.processedImage = null;
    this.isProcessing = false;
    
    // 重置UI
    document.getElementById('original-placeholder').style.display = 'flex';
    document.getElementById('result-placeholder').style.display = 'flex';
    document.getElementById('original-canvas').style.display = 'none';
    document.getElementById('result-canvas').style.display = 'none';
    document.getElementById('file-input').value = '';
    
    this.updateUI();
  }

  updateUI() {
    const hasFile = !!this.currentFile;
    const hasResult = !!this.processedImage;
    
    // 处理按钮
    document.getElementById('process-btn').disabled = !hasFile || this.isProcessing;
    
    // 下载按钮
    document.getElementById('download-btn').disabled = !hasResult;
    
    // 重置按钮
    document.getElementById('reset-btn').disabled = !hasFile && !hasResult;
  }

  // UI辅助方法
  showLoading(message) {
    // 实现加载状态显示
  }

  hideLoading() {
    // 隐藏加载状态
  }

  showProgress(message, percent) {
    // 显示进度条
  }

  updateProgress(percent) {
    // 更新进度
  }

  hideProgress() {
    // 隐藏进度条
  }

  showSuccess(message) {
    // 显示成功提示
  }

  showError(message) {
    // 显示错误提示
  }
}

// 启动应用
window.addEventListener('DOMContentLoaded', () => {
  window.app = new QuickBGApp();
});
```

### 错误分类与处理

#### 1. 用户输入错误
```javascript
const ERROR_MESSAGES = {
  FILE_TYPE: '请选择 JPG、PNG 或 WebP 格式的图片',
  FILE_SIZE: '图片大小不能超过 5MB',
  NO_FILE: '请先选择图片',
  NETWORK: '网络连接失败，请检查网络后重试',
  API_LIMIT: 'API调用次数已达上限',
  SERVER_ERROR: '服务器处理失败，请稍后重试',
  UNKNOWN: '发生未知错误，请刷新页面重试',
};

function handleError(error) {
  let message = ERROR_MESSAGES.UNKNOWN;
  
  if (error.message.includes('format')) {
    message = ERROR_MESSAGES.FILE_TYPE;
  } else if (error.message.includes('size')) {
    message = ERROR_MESSAGES.FILE_SIZE;
  } else if (error.message.includes('network')) {
    message = ERROR_MESSAGES.NETWORK;
  } else if (error.message.includes('limit') || error.message.includes('credit')) {
    message = ERROR_MESSAGES.API_LIMIT;
  } else if (error.message.includes('server')) {
    message = ERROR_MESSAGES.SERVER_ERROR;
  }
  
  return message;
}
```

#### 2. API错误处理
```javascript
class APIErrorHandler {
  static async handleResponse(response) {
    if (response.ok) {
      return response;
    }
    
    const status = response.status;
    
    switch (status) {
      case 400:
        throw new Error('请求参数错误');
      case 401:
        throw new Error('API密钥无效');
      case 402:
        throw new Error('API额度不足');
      case 429:
        throw new Error('请求过于频繁，请稍后重试');
      case 500:
        throw new Error('服务器内部错误');
      case 503:
        throw new Error('服务暂时不可用');
      default:
        throw new Error(`请求失败 (${status})`);
    }
  }
  
  static async withRetry(fn, maxRetries = 3) {
    let lastError;
    
    for (let i = 0; i < maxRetries; i++) {
      try {
        return await fn();
      } catch (error) {
        lastError = error;
        
        // 如果是网络错误，等待后重试
        if (error.message.includes('network') || error.message.includes('timeout')) {
          await this.delay(1000 * Math.pow(2, i)); // 指数退避
          continue;
        }
        
        // 其他错误直接抛出
        throw error;
      }
    }
    
    throw lastError;
  }
  
  static delay(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```
